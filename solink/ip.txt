        function addSpan(type, message, style) {
            style = style || "";
            return "<span class='label label-"+type+"' style='"+style+"'>"+ message +"</span>";
        }

        function formatIPv4Data(resp) {
            var $p = $("#v4_conn");
            $p.find(".proto_address, .proto_hostname, .proto_isp").empty();

            $p.find(".proto_supported").html(addSpan('success', "Supported", "font-size:1em;"));

            let link = "<a href='https://ip.sb/whois/"+resp.address+"'>"+resp.address+"</a>";
            let pop = '<span id="ip2long" data-toggle="popover" data-trigger="hover" data-content="'+ resp.address + ' => ' + resp.ip2long +'" data-title="IP2Long">'+ link +'</span>';

            $p.find(".proto_address").html(link);
            //$('#ip2long').popover({
            //    delay: { "hide": 3000 }
            //});

            $p.find(".proto_hostname").html(resp.hostname ? resp.hostname : "<span class='label label-warning'>None</span>");
            $p.find(".proto_location").html((resp.city ? resp.city : "") + ' ' + (resp.province ? resp.province + ', ' : "") + resp.country);
            $p.find(".proto_isp").html(resp.isp.name ? resp.isp.name : addSpan('warning', "Unknown"));
            $p.find(".proto_organization").html(resp.isp.organization ? resp.isp.organization : addSpan('warning', "Unknown"));
            $p.find(".proto_asn").html(resp.isp.ASN ? "<a href='https://ip.sb/whois/AS"+ resp.isp.ASN +"'>AS"+ resp.isp.ASN +"</a>" : addSpan('warning', "Unknown"));
            $p.find(".proto_aso").html(resp.isp.ASO ? resp.isp.ASO : addSpan('warning', "Unknown"));
        }
        function formatIPv6Data(resp) {
            var $p = $("#v6_conn");
            $p.find(".proto_address, .proto_hostname, .proto_isp, .proto_country").empty();

            $p.find(".proto_supported").html(addSpan('success', "Supported", "font-size:1em;"));
            $p.find(".proto_address").html("<a href='/whois/"+resp.address+"'>"+resp.address+"</a>");
            $p.find(".proto_hostname").html(resp.hostname ? resp.hostname : addSpan('warning', "None"));
            $p.find(".proto_country").html(resp.country ? resp.country : addSpan('warning', "Unknown"));
            $p.find(".proto_organization").html(resp.isp.organization ? resp.isp.organization : addSpan('warning', "Unknown"));
            $p.find(".proto_asn").html(resp.isp.ASN ? "<a href='https://ip.sb/whois/AS"+ resp.isp.ASN +"'>AS"+ resp.isp.ASN +"</a>" : addSpan('warning', "Unknown"));
            $p.find(".proto_aso").html(resp.isp.ASO ? resp.isp.ASO : addSpan('warning', "Unknown"));
        }

        function hideIPv6Area() {
            var $p = $("#v6_conn");
            $p.find(".proto_address, .proto_hostname, .proto_isp, .proto_country").empty();

            $p.find(".proto_address").parent().hide();
            $p.find(".proto_hostname").parent().hide();
            $p.find(".proto_country").parent().hide();
            $p.find(".proto_organization").parent().hide();
            $p.find(".proto_asn").parent().hide();
            $p.find(".proto_aso").parent().hide();
        }

        function test_v4_conn() {
            var $p = $("#v4_conn");
            $p.find(".proto_address, .proto_hostname, .proto_isp").empty();
            $.ajax({
                url: "https://ipv4.ip.sb/addrinfo",
                dataType: "jsonp",
                timeout: 15000,
                success: function(resp) {
                    formatIPv4Data(resp);
                },
                beforeSend: function() {
                    $p.find(".proto_supported").html(addSpan('info', "Checking", "font-size:1em;"));
                },
                error: function(xopt, err) {
                    $p.find(".proto_supported").html(addSpan('danger', "Not Supported", "font-size:1em;"));
                },
                complete: function(){
                },
            });
        }

                },
